program BubbleSort;
var
  arr: array[1..100] of integer;
  i, j, temp, sizeOfArray: integer;
begin
  writeln('Enter the size of the array: ');
  read(sizeOfArray);
  for i := 1 to sizeOfArray do
  begin
    write('Element ', i, ': ');
    readln(arr[i]);
  end;

  for i := 1 to sizeOfArray do
  begin
    for j := 1 to sizeOfArray - i do
    begin
      if arr[j] > arr[j + 1] then
      begin
        temp := arr[j];
        arr[j] := arr[j + 1];
        arr[j + 1] := temp;
      end;
    end;
  end;    

  writeln('Sorted array:');
  for i := 1 to sizeOfArray do
  begin
    writeln(arr[i]);
  end;    
end.
